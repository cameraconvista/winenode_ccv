import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';

const ManualWineInsertPage = () => {
  const navigate = useNavigate();
  const [categoria, setCategoria] = useState('');
  const [fornitore, setFornitore] = useState('');
  const [elencoVini, setElencoVini] = useState('');
  const [inseriti, setInseriti] = useState(0);
  const fornitori = ['Seleziona fornitore', 'Fornitore A', 'Fornitore B', 'Fornitore C'];
  const categorie = ['Rosso', 'Bianco', 'Bollicine', 'Rosato'];

  useEffect(() => {
    const esistenti = JSON.parse(localStorage.getItem('vini') || '[]');
    setInseriti(esistenti.length);
  }, []);

  const salvaVini = (sostituisci = false) => {
    if (!categoria || elencoVini.trim() === '') {
      alert('Inserisci almeno una categoria e un elenco vini valido.');
      return;
    }
    const nuovi = elencoVini
      .split('\n')
      .map((riga) => riga.trim())
      .filter((r) => r !== '')
      .map((nome) => ({
        nome,
        tipo: categoria,
        fornitore: fornitore || '',
        costo: 0,
        soglia_minima: 0,
      }));

    const esistenti = JSON.parse(localStorage.getItem('vini') || '[]');
    const finali = sostituisci ? nuovi : [...esistenti, ...nuovi];

    localStorage.setItem('vini', JSON.stringify(finali));
    setInseriti(finali.length);
    setElencoVini('');
    alert(`${nuovi.length} vini ${sostituisci ? 'sostituiti' : 'aggiunti'} nella categoria ${categoria.toUpperCase()}`);
  };

  return (
    <div className="p-4 max-w-xl mx-auto text-white">
      <h1 className="text-center text-2xl font-bold mb-2">üç∑ INSERIMENTO MANUALE VINI</h1>
      <p className="text-center mb-4">Aggiungi i tuoi vini incollandoli sotto</p>
      <div className="bg-[#2b1c1c] rounded-xl p-4 space-y-4">
        <div className="text-center text-md bg-[#1c1212] p-2 rounded">Hai inserito <span className="font-bold text-yellow-400">{inseriti}</span> vini finora</div>

        <label className="block text-sm font-bold mt-2">Categoria Vino *</label>
        <select
          className="w-full p-2 rounded bg-[#1e1e1e]"
          value={categoria}
          onChange={(e) => setCategoria(e.target.value)}
        >
          <option value="">Seleziona categoria</option>
          {categorie.map((c) => (
            <option key={c} value={c}>{c}</option>
          ))}
        </select>

        <label className="block text-sm font-bold mt-2">Fornitore (opzionale)</label>
        <select
          className="w-full p-2 rounded bg-[#1e1e1e]"
          value={fornitore}
          onChange={(e) => setFornitore(e.target.value)}
        >
          {fornitori.map((f, i) => (
            <option key={i} value={i === 0 ? '' : f}>{f}</option>
          ))}
        </select>

        <label className="block text-sm font-bold mt-2">Elenco Vini (uno per riga)</label>
        <textarea
          rows={10}
          className="w-full p-2 rounded bg-[#1e1e1e]"
          placeholder={`Es.\nBarolo DOCG 2021\nChianti Classico Riserva\nEtna Rosso 2020`}
          value={elencoVini}
          onChange={(e) => setElencoVini(e.target.value)}
        ></textarea>

        <div className="flex flex-col gap-2 mt-4">
          <button
            className="bg-green-600 hover:bg-green-700 p-3 rounded text-white font-bold"
            onClick={() => salvaVini(false)}
          >
            ‚ûï Aggiungi a lista esistente
          </button>
          <button
            className="bg-red-600 hover:bg-red-700 p-3 rounded text-white font-bold"
            onClick={() => {
              if (confirm('‚ö†Ô∏è Vuoi davvero sostituire l‚Äôintera lista vini esistente?')) {
                salvaVini(true);
              }
            }}
          >
            üóë Sostituisci lista esistente
          </button>
        </div>

        <p className="text-center text-xs mt-4 text-gray-400">
          I vini vengono salvati nel browser. Sincronizzazione futura prevista.
        </p>
      </div>
    </div>
  );
};

export default ManualWineInsertPage;
